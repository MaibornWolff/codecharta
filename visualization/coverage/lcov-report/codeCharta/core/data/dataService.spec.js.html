        //inputs
        let a = {"a":100,"b":10,"c":1};
        let b = {"a":110,"b":11,"c":0};
        let c = {"a":110,"b":11,"c":0, "d":10};

        let e = {"d":110,"e":11};
 
        // system under test
        let sut = dataService;
 
        // assertion
        let ab = sut.calculateAttributeListDelta(a,b);
        expect(ab.a).to.equal(b.a-a.a);
